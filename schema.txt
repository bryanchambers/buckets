CREATE TABLE user(
    id       INTEGER PRIMARY KEY,
    username TEXT,
    password TEXT
);

CREATE TABLE bucket(
    id      INTEGER PRIMARY KEY, 
    name    TEXT,
    balance INTEGER,
    refill  INTEGER,
    size    INTEGER
);

CREATE TABLE purchase(
    id        INTEGER PRIMARY KEY,
    bucket_id INTEGER,
    desc      TEXT,
    amount    INTEGER,
    date      INTEGER
);
